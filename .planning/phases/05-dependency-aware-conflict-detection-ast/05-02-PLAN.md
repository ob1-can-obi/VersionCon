Stand up the tier-1 grammars and adapters for JavaScript + TypeScript (the project's own language — every Phase 5 dogfood test will land on this path):

1. **`src/ast/grammars.ts`** — single owner of `Parser.init()` (must be called exactly once per process), `Language.load()` with a memoized cache (loading the same WASM twice leaks memory), and the locateFile resolver that points web-tree-sitter at `dist/vendor/tree-sitter/` so the WASM runtime can find the language binaries at runtime. Also exports `getParserFor(languageId)`: returns a Parser pre-configured with the right Language (one Parser per languageId, reused across files in the worker).

2. **`src/ast/adapters/javascript.ts`** — `JavaScriptAdapter` walks the tree-sitter syntax tree once per file via `tree.walk()` cursor (the SPEC-recommended traversal) and emits the SymbolIndex + ReferenceIndex. Covers the AST nodes that matter for v1: `function_declaration` / `arrow_function` (only when assigned to a named binding) → `SymbolIndex.functions`; `class_declaration` → `SymbolIndex.classes`; `method_definition` inside classes → `SymbolIndex.functions` (with `name = ClassName.methodName`); `lexical_declaration` (const/let) + `variable_declaration` (var) → `SymbolIndex.variables`; `import_statement` → `SymbolIndex.imports`; `export_statement` → `SymbolIndex.exports`; `call_expression` → `ReferenceIndex.calls`; bare `identifier` reads at top-level or within function bodies → `ReferenceIndex.reads`.

3. **`src/ast/adapters/typescript.ts`** — `TypeScriptAdapter` composes the JS extractor (TS is JS plus types). Adds: `interface_declaration` → `SymbolIndex.classes` (interfaces ARE the contract a TS consumer depends on, so treat them as classes for v1 attribution); `type_alias_declaration` → SKIPPED (types don't drive runtime calls); `function_signature` (declaration-only signatures in `.d.ts`) → `SymbolIndex.functions`. **TSX routing:** when `relativePath` ends with `.tsx`, parse with the `tsx` grammar instead of the `typescript` grammar (web-tree-sitter ships them as two distinct WASM files in tree-sitter-typescript).

4. **WASM vendoring** — drop three files into `src/vendor/tree-sitter/`:

   - `tsx.wasm` from `tree-sitter-typescript@0.23.2` (`node_modules/tree-sitter-typescript/tree-sitter-tsx.wasm`)

1. Create `src/ast/grammars.ts`:
   ```typescript
   import * as path from 'node:path';
   import { Parser, Language } from 'web-tree-sitter';
   import type { LanguageId } from './types.js';

   /**
    * Phase 5 Plan 05-02 — grammar loader.
    *
    * Single owner of:
    *   - Parser.init() (must be called exactly once per process)
    *   - Language.load() with a per-LanguageId cache (loading the same WASM
    *     twice leaks ~300KB of Wasm linear memory per leak; cache hard)
    *   - locateFile resolver that points web-tree-sitter at dist/vendor/tree-sitter/
    *     so the worker (Wave 4, spawned via child_process.fork) finds the
    *     runtime engine + grammar binaries
    *
    * T-05-01 mitigation: Parser.init failure throws synchronously — caller in
    * Wave 4 catches and re-forks. Once init succeeds, getParserFor(id) is safe
    * to call repeatedly.
    */

   let initPromise: Promise<void> | null = null;
   const LANGUAGE_CACHE = new Map<LanguageId, Language>();
   const PARSER_CACHE = new Map<LanguageId, Parser>();

   /** Resolve a grammar WASM path. .tsx routes to the tsx grammar variant. */
   function wasmPathFor(languageId: LanguageId, relativePath?: string): string {
     // Distinguish .tsx vs .ts at the grammar level. The TS grammar can parse .ts
     // but NOT .tsx; the TSX grammar handles both, so .tsx files MUST route
     // through the tsx WASM. The caller (Wave 2 adapter) passes relativePath
     // when routing is path-dependent.
     if (languageId === 'typescript' && relativePath?.endsWith('.tsx')) {
       return path.join(distVendorDir(), 'tsx.wasm');
     }
     const fileName = languageId === 'typescript' ? 'typescript.wasm' : `${languageId}.wasm`;
     return path.join(distVendorDir(), fileName);
   }

   /** Compute the dist/vendor/tree-sitter/ absolute path. Works from both
    * dist/extension.js (host process) and dist/ast-worker.js (Wave 4 worker). */
   function distVendorDir(): string {
     return path.resolve(__dirname, 'vendor', 'tree-sitter');
   }

   export async function initParser(): Promise<void> {
     if (initPromise) return initPromise;
     initPromise = Parser.init({
       locateFile: (file: string) => path.join(distVendorDir(), file),
     });
     return initPromise;
   }

   export async function loadGrammar(
     languageId: LanguageId,
     relativePath?: string,
   ): Promise<Language> {
     await initParser();
     // For .tsx use a distinct cache key so .ts and .tsx do not share a Language.
     const cacheKey = (languageId === 'typescript' && relativePath?.endsWith('.tsx'))
       ? ('typescript-tsx' as LanguageId)  // synthetic cache key, not exposed
       : languageId;
     let lang = LANGUAGE_CACHE.get(cacheKey);
     if (!lang) {
       lang = await Language.load(wasmPathFor(languageId, relativePath));
       LANGUAGE_CACHE.set(cacheKey, lang);
     }
     return lang;
   }

   export async function getParserFor(
     languageId: LanguageId,
     relativePath?: string,
   ): Promise<Parser> {
     await initParser();
     const cacheKey = (languageId === 'typescript' && relativePath?.endsWith('.tsx'))
       ? ('typescript-tsx' as LanguageId)
       : languageId;
     let parser = PARSER_CACHE.get(cacheKey);
     if (!parser) {
       const language = await loadGrammar(languageId, relativePath);
       parser = new Parser();
       parser.setLanguage(language);
       PARSER_CACHE.set(cacheKey, parser);
     }
     return parser;
   }

   /** Test-only reset. */
   export function _resetGrammarsForTests(): void {
     initPromise = null;
     LANGUAGE_CACHE.clear();
     for (const p of PARSER_CACHE.values()) p.delete();
     PARSER_CACHE.clear();
   }
   ```

           if (ident) out.functions.push({ name: ident.text, line: ident.startPosition.row + 1 });

           // Recurse INTO class_body with classCtx = ident.text — see executor note below
           break;

           } else if (ident) {
             out.functions.push({ name: ident.text, line: ident.startPosition.row + 1 });
           }
           break;
         }
         case 'lexical_declaration':
         case 'variable_declaration': {
           // Each declarator child contributes a variable (or a function if
           // its value is an arrow_function).
           for (let i = 0; i < n.namedChildCount; i++) {
             const decl = n.namedChild(i);
             if (!decl) continue;
             const name = decl.childForFieldName('name');
             const value = decl.childForFieldName('value');
             if (!name) continue;
             if (value && (value.type === 'arrow_function' || value.type === 'function_expression')) {
               out.functions.push({ name: name.text, line: name.startPosition.row + 1 });
             } else {
               out.variables.push({ name: name.text, line: name.startPosition.row + 1 });
             }
           }
           break;
         }
         case 'import_statement': {
           const source = n.childForFieldName('source');
           const fromPath = source ? source.text.slice(1, -1) : '';  // strip quotes
           // Walk import_clause children for default/named/namespace imports.
           // Each emits one entry with from=fromPath, line=identifier line.
           const importClause = n.namedChild(0);
           if (importClause) {
             for (let i = 0; i < importClause.namedChildCount; i++) {
               const c = importClause.namedChild(i);
               if (!c) continue;
               // Default import: identifier directly. Named: named_imports → import_specifier(s).
               // Namespace: namespace_import → identifier child.
               if (c.type === 'identifier') {
                 out.imports.push({ name: c.text, from: fromPath, line: c.startPosition.row + 1 });
               } else if (c.type === 'named_imports') {
                 for (let j = 0; j < c.namedChildCount; j++) {
                   const spec = c.namedChild(j);
                   if (!spec) continue;
                   const local = spec.childForFieldName('alias') ?? spec.childForFieldName('name') ?? spec.namedChild(0);
                   if (local) out.imports.push({ name: local.text, from: fromPath, line: local.startPosition.row + 1 });
                 }
               } else if (c.type === 'namespace_import') {
                 const local = c.namedChild(0);
                 if (local) out.imports.push({ name: local.text, from: fromPath, line: local.startPosition.row + 1 });
               }
             }
           }
           break;
         }
         case 'export_statement': {
           // export const x = 1  → variables: x AND exports: x
           // export function f() → functions: f AND exports: f
           // export default ...  → exports: 'default'
           // export { foo } from './bar' → re-export; emit name 'foo' in imports too (treat as both)
           // Executor: walk export_statement children. The exported declaration is a function_declaration,
           // class_declaration, or lexical_declaration child whose names are handled by the recursive
           // visit pass — DO NOT double-emit. For exports: walk the export_clause's specifiers and
           // emit one export entry per name. For default exports, emit { name: 'default', line: row+1 }.
           // See behavior tests 9-11 for the canonical output.
           break;
         }
       }
     }
   }
