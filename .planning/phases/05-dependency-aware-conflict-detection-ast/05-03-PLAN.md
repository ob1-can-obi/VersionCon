- assignment: `x = 1`                    — left side is identifier or expression_list

1. Create `src/ast/adapters/python.ts`:
   ```typescript
   import type { Parser, Node } from 'web-tree-sitter';
   import type { AstAdapter, LanguageId, SymbolIndex, ReferenceIndex } from '../types.js';
   import { getParserFor } from '../grammars.js';
   import { registerAdapter } from '../AstFactory.js';

   /**
    * Phase 5 Plan 05-03 — tier-1 Python adapter (SC-3).
    *
    * Same contract as JavaScriptAdapter: synchronous extract* methods, crash-tolerant
    * try/catch wrapping (T-05-01), no fs / no IO. The async work (parser init +
    * grammar load) happens via prepare(); the worker awaits before invocation.
    *
    * Module-scope discipline: only top-level assignments emit SymbolIndex.variables.
    * Function-local bindings are not symbols at the v1 attribution layer (SC-1 says
    * "you call this function in line 34" — the smallest unit attributable to a
    * member is a top-level binding or class method, not a function-local). Tests
    * pin this invariant.
    */
   export class PythonAdapter implements AstAdapter {
     readonly languageId: LanguageId = 'python';
     private parser: Parser | null = null;

     async prepare(): Promise<void> {
       if (!this.parser) this.parser = await getParserFor('python');
     }

     extractSymbols(source: string, relativePath: string): SymbolIndex {
       void relativePath;
       const empty: SymbolIndex = { functions: [], classes: [], variables: [], imports: [], exports: [] };
       if (!this.parser) return empty;
       try {
         const tree = this.parser.parse(source);
         if (!tree) return empty;
         return this.walkSymbols(tree.rootNode);
       } catch {
         return empty;
       }
     }

     extractReferences(source: string, relativePath: string): ReferenceIndex {
       void relativePath;
       const empty: ReferenceIndex = { calls: [], reads: [], imports: [] };
       if (!this.parser) return empty;
       try {
         const tree = this.parser.parse(source);
         if (!tree) return empty;
         return this.walkReferences(tree.rootNode);
       } catch {
         return empty;
       }
     }

     /** Module-scope walker — recurses INTO class_definition (for methods) but NOT into function_definition (for local vars). */
     private walkSymbols(root: Node): SymbolIndex {
       const out: SymbolIndex = { functions: [], classes: [], variables: [], imports: [], exports: [] };
       const visit = (n: Node, classCtx: string | null) => {
         switch (n.type) {
           case 'function_definition': {
             const ident = n.childForFieldName('name');
             if (ident) {
               const name = classCtx ? `${classCtx}.${ident.text}` : ident.text;
               out.functions.push({ name, line: ident.startPosition.row + 1 });
             }
             // DO NOT recurse into function bodies — local vars are not symbols.
             return;
           }
           case 'class_definition': {
             const ident = n.childForFieldName('name');
             if (ident) out.classes.push({ name: ident.text, line: ident.startPosition.row + 1 });
             // Recurse into the class body so methods are emitted with the class prefix.
             const body = n.childForFieldName('body');
             if (body) {
               for (let i = 0; i < body.namedChildCount; i++) {
                 const c = body.namedChild(i);
                 if (c) visit(c, ident?.text ?? null);
               }
             }
             return;
           }
           case 'decorated_definition': {
             // The 'definition' field holds the wrapped function/class.
             const inner = n.childForFieldName('definition');
             if (inner) visit(inner, classCtx);
             return;
           }
           case 'assignment': {
             // Module-scope only — caller (top-level visit loop below) ensures this.
             const lhs = n.childForFieldName('left');
             if (lhs) this.emitAssignmentNames(lhs, out);
             return;
           }
           case 'import_statement': {
             // `import os, sys as s` — children are dotted_name or aliased_import.
             for (let i = 0; i < n.namedChildCount; i++) {
               const c = n.namedChild(i);
               if (!c) continue;
               if (c.type === 'aliased_import') {
                 const orig = c.childForFieldName('name');
                 const alias = c.childForFieldName('alias');
                 if (orig && alias) out.imports.push({ name: alias.text, from: orig.text, line: alias.startPosition.row + 1 });
               } else if (c.type === 'dotted_name') {
                 out.imports.push({ name: c.text, from: c.text, line: c.startPosition.row + 1 });
               }
             }
             return;
           }
           case 'import_from_statement': {
             const moduleName = n.childForFieldName('module_name');
             const fromPath = moduleName?.text ?? '';
             for (let i = 0; i < n.namedChildCount; i++) {
               const c = n.namedChild(i);
               if (!c || c === moduleName) continue;
               if (c.type === 'aliased_import') {
                 const orig = c.childForFieldName('name');
                 const alias = c.childForFieldName('alias');
                 if (orig && alias) out.imports.push({ name: alias.text, from: fromPath, line: alias.startPosition.row + 1 });
                 else if (orig) out.imports.push({ name: orig.text, from: fromPath, line: orig.startPosition.row + 1 });
               } else if (c.type === 'dotted_name') {
                 out.imports.push({ name: c.text, from: fromPath, line: c.startPosition.row + 1 });
               }
             }
             return;
           }
         }
       };
       // Top-level: iterate module's named children directly so we never recurse
       // into function bodies for assignments.
       for (let i = 0; i < root.namedChildCount; i++) {
         const c = root.namedChild(i);
         if (c) visit(c, null);
       }
       return out;
     }

     /** Emits SymbolIndex.variables names from an LHS node (identifier / pattern_list / expression_list). */
     private emitAssignmentNames(lhs: Node, out: SymbolIndex): void {
       if (lhs.type === 'identifier') {
         out.variables.push({ name: lhs.text, line: lhs.startPosition.row + 1 });
       } else if (lhs.type === 'pattern_list' || lhs.type === 'expression_list' || lhs.type === 'tuple_pattern') {
         for (let i = 0; i < lhs.namedChildCount; i++) {
           const c = lhs.namedChild(i);
           if (c) this.emitAssignmentNames(c, out);
         }
       }
       // Attribute LHS (`self.x = …`) intentionally skipped — class field
       // detection is deferred (5.x). Pattern subscript LHS (`a[0] = …`) is
       // not a name binding either.
     }

     private walkReferences(root: Node): ReferenceIndex {
       const out: ReferenceIndex = { calls: [], reads: [], imports: [] };
       // Walk EVERY descendant (call_expression and attribute can appear at
       // any depth). Use cursor-based DFS like JavaScriptAdapter.walkReferences.
       const cursor = root.walk();
       const visit = () => {
         const n = cursor.currentNode;
         switch (n.type) {
           case 'call': {
             const fn = n.childForFieldName('function');
             if (fn) {
               // identifier → bare name; attribute → dotted chain. Use n.text
               // trimmed to the function's text for the dotted form.
               out.calls.push({ name: fn.text, line: fn.startPosition.row + 1 });
             }
             break;
           }
           case 'import_statement':
           case 'import_from_statement':
             // Captured separately via extractSymbols; references treats them as
             // references too so the join can resolve "you import foo".
             // Mirror the symbol-side import emission. Executor: factor the
             // emit into a shared private helper called from both walkers.
             break;
         }
         if (cursor.gotoFirstChild()) {
           do { visit(); } while (cursor.gotoNextSibling());
           cursor.gotoParent();
         }
       };
       visit();
       return out;
     }
   }

   registerAdapter('python', new PythonAdapter());
   ```

1. Create `src/ast/adapters/fallback.ts`:
   ```typescript
   import type { AstAdapter, LanguageId, SymbolIndex, ReferenceIndex } from '../types.js';

   /**
    * Phase 5 Plan 05-03 — line-level fallback (SC-3 / CONF-10).
    *
    * Used when:
    *   - The language has no real grammar (java / cpp v1)
    *   - The grammar fails to parse the source (try/catch fallthrough)
    *   - The file is over 500KB (skipPolicy excludes before this adapter ever sees it)
    *
    * Contract:
    *   - SymbolIndex.variables emits ONE entry per non-blank, non-comment line.
    *     name = first whitespace-delimited token of the line (after trimming).
    *     line = 1-based row.
    *   - All other SymbolIndex arrays are empty.
    *   - ReferenceIndex is always entirely empty — line-level fallback cannot
    *     derive references, and the join layer treats absence-of-references
    *     as "file-level only, no symbol match attempted" (CONF-10 file-level
    *     fallback).
    *
    * The "first-token-per-line" emission is NOT meaningful as a symbol name —
    * it exists so the Wave 4 join has SOMETHING per line to fingerprint. When
    * a fallback file has multiple members with overlapping content, the join
    * can still report file-level impact via affectsLocal computed from
    * file-paths (mirrors Phase 4 CONF-07 behavior). Wave 4 join layer MUST NOT
    * cross-attribute fallback variables to real-grammar symbols.
    */
   export class FallbackAdapter implements AstAdapter {
     constructor(public readonly languageId: LanguageId) {}

     extractSymbols(source: string, relativePath: string): SymbolIndex {
       void relativePath;
       const out: SymbolIndex = { functions: [], classes: [], variables: [], imports: [], exports: [] };
       if (typeof source !== 'string' || source.length === 0) return out;
       // Strip BOM
       const cleaned = source.charCodeAt(0) === 0xFEFF ? source.slice(1) : source;
       const lines = cleaned.split('\n');
       for (let i = 0; i < lines.length; i++) {
         const raw = lines[i].replace(/\r$/, '');
         const trimmed = raw.trim();
         if (trimmed.length === 0) continue;                       // blank
         if (/^(\/\/|#|\/\*)/.test(trimmed)) continue;             // comment (best-effort)
         const firstToken = trimmed.split(/\s+/, 1)[0];
         if (!firstToken) continue;
         out.variables.push({ name: firstToken, line: i + 1 });
       }
       return out;
     }

     extractReferences(source: string, relativePath: string): ReferenceIndex {
       void source;
       void relativePath;
       return { calls: [], reads: [], imports: [] };
     }
   }

   export function createFallbackAdapter(languageId: LanguageId): FallbackAdapter {
     return new FallbackAdapter(languageId);
   }
   ```
