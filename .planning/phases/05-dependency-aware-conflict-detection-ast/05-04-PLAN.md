1. Create `src/ast/joinImpact.ts`:
   ```typescript
   import type { SymbolIndex, ReferenceIndex, AffectedSymbol, AnalysisResult } from './types.js';

   /**
    * Phase 5 Plan 05-04 — pure-fn symbol/reference join.
    *
    * v1 scope decisions (T-05-05 mitigation):
    *   - Cross-file name collision avoided by gating on imports: a member's
    *     reference to `foo` only counts as a caller of `cart-helpers.ts`'s foo
    *     if the member's reference-file imports `foo` from `cart-helpers`.
    *   - Symbol "modified" detection compares NAME + LINE only — value changes
    *     are not detected (v1 SymbolIndex has no value/hash field). 5.x can
    *     extend SymbolIndex with `hash?: string` and surface value diffs.
    *   - Symbol kinds derived by which array of SymbolIndex emitted it:
    *     functions → 'function', classes → 'class', variables → 'variable',
    *     imports → 'import', exports → 'export'.
    *   - Determinism: result.affectedSymbols sorted by (changedIn, name); callers within
    *     each AffectedSymbol sorted by displayName.
    */

   interface ChangedSymbol {
     name: string;
     kind: AffectedSymbol['kind'];
     changedIn: string;        // file path
   }

   export function joinImpact(
     changedSymbolsPerFile: Map<string, { pre: SymbolIndex; post: SymbolIndex }>,
     memberReferences: Map<string, Map<string, ReferenceIndex>>,
     memberDisplayNames: Map<string, string>,
     unsupportedLanguages: string[] = [],
   ): AnalysisResult {
     // 1. Compute the union changed-symbol list across all files.
     const changes: ChangedSymbol[] = [];
     for (const [file, { pre, post }] of changedSymbolsPerFile) {
       diffArray(pre.functions, post.functions, 'function', file, changes);
       diffArray(pre.classes,   post.classes,   'class',    file, changes);
       diffArray(pre.variables, post.variables, 'variable', file, changes);
       diffArray(pre.imports,   post.imports,   'import',   file, changes);
       diffArray(pre.exports,   post.exports,   'export',   file, changes);
     }

     // 2. For each change, find callers across all members.
     const affected: AffectedSymbol[] = [];
     for (const change of changes) {
       const callers: AffectedSymbol['callers'] = [];
       for (const [memberId, fileMap] of memberReferences) {
         for (const [refFile, refs] of fileMap) {
           // T-05-05 file-scoped match: require the ref-file to import from change.changedIn.
           const importsChangeFile = refs.imports.some(imp =>
             pathMatches(imp.from, change.changedIn) && imp.name === change.name);
           if (importsChangeFile) {
             // The import statement itself is a caller — record its line.
             const importLine = refs.imports.find(imp =>
               pathMatches(imp.from, change.changedIn) && imp.name === change.name)!.line;
             callers.push({
               memberId,
               displayName: memberDisplayNames.get(memberId) ?? 'Unknown',
               file: refFile,
               line: importLine,
             });
             // Then add each call/read of the changed symbol's name within this file.
             for (const c of refs.calls) {
               if (extractBaseName(c.name) === change.name) {
                 callers.push({
                   memberId, displayName: memberDisplayNames.get(memberId) ?? 'Unknown',
                   file: refFile, line: c.line,
                 });
               }
             }
             for (const r of refs.reads) {
               if (r.name === change.name) {
                 callers.push({
                   memberId, displayName: memberDisplayNames.get(memberId) ?? 'Unknown',
                   file: refFile, line: r.line,
                 });
               }
             }
           }
         }
       }
       if (callers.length > 0) {
         callers.sort((a, b) => a.displayName.localeCompare(b.displayName) || a.line - b.line);
         affected.push({ name: change.name, kind: change.kind, changedIn: change.changedIn, callers });
       }
     }

     // 3. Determinism: sort.
     affected.sort((a, b) =>
       a.changedIn.localeCompare(b.changedIn) || a.name.localeCompare(b.name),
     );

     // 4. perMember regroup.
     const perMember: Record<string, AffectedSymbol[]> = {};
     for (const sym of affected) {
       for (const c of sym.callers) {
         if (!perMember[c.memberId]) perMember[c.memberId] = [];
         // Avoid double-adding the same symbol entry for the same member.
         if (!perMember[c.memberId].some(s => s.name === sym.name && s.changedIn === sym.changedIn)) {
           perMember[c.memberId].push(sym);
         }
       }
     }
     return { affectedSymbols: affected, perMember, unsupportedLanguages };
   }

   function diffArray(
     pre: Array<{ name: string; line: number }>,
     post: Array<{ name: string; line: number }>,
     kind: AffectedSymbol['kind'],
     file: string,
     out: ChangedSymbol[],
   ): void {
     const preMap = new Map(pre.map(p => [p.name, p.line]));
     const postMap = new Map(post.map(p => [p.name, p.line]));
     for (const name of new Set([...preMap.keys(), ...postMap.keys()])) {
       const preLine = preMap.get(name);
       const postLine = postMap.get(name);
       const added = preLine === undefined && postLine !== undefined;
       const removed = preLine !== undefined && postLine === undefined;
       const modified = preLine !== undefined && postLine !== undefined && preLine !== postLine;
       if (added || removed || modified) {
         out.push({ name, kind, changedIn: file });
       }
     }
   }

   /** True if import-from path resolves to the changed file (best-effort v1 — string compare with optional /index.ts and extension drop). */
   function pathMatches(importFrom: string, changedIn: string): boolean {
     // Handle './foo' importing from 'src/foo.ts' / 'src/foo/index.ts' — strip leading ./ and trailing extension
     const normalizeFor = (p: string) => p.replace(/^\.\//, '').replace(/\.(ts|tsx|js|jsx|mjs|cjs|py|java|cpp|cc|cxx|h|hpp)$/i, '').replace(/\/index$/, '');
     return normalizeFor(importFrom) === normalizeFor(changedIn);
   }

   function extractBaseName(callName: string): string {
     // 'obj.foo' → 'foo' for v1 cross-attribution (file-scoped, so safe).
     const idx = callName.lastIndexOf('.');
     return idx < 0 ? callName : callName.slice(idx + 1);
   }
   ```

    * Phase 5 Plan 05-04 — forked AST worker.
    *
    * Runs in a child_process.fork()'d Node process. MUST NOT import vscode.
    *
    * Lifecycle:
    *   - Boot: Parser.init via grammars.ts, side-effect import of all 5 adapters
    *     (registerAdapter populates the factory registry).
    *   - Per message: parse changedFiles + memberTrackedFiles, join, respond.
    *
    * Crash tolerance:
    *   - Per-file try/catch around adapter calls — one bad file does not poison
    *     the whole pass.
    *   - On worker.send failure: log + exit(1); parent re-forks (Plan 05-04
    *     Task 3 owns parent-side recovery).
    *   - process.on('uncaughtException', err => { process.send({...errorResp...}); process.exit(1); })
    *     so any escaped throw cleanly responds before exit, parent re-forks
    *     on next request.

   import { getAdapter, detectLanguageFromPath } from './AstFactory.js';
   import { shouldSkip } from './skipPolicy.js';
   import { joinImpact } from './joinImpact.js';
   import type { AnalyzePayload, AnalysisResponse, SymbolIndex, ReferenceIndex, LanguageId } from './types.js';
   import type { AstAdapter } from './types.js';

   // Side-effect imports trigger registerAdapter for each adapter.
   import './adapters/javascript.js';
   import './adapters/typescript.js';
   import './adapters/python.js';
   import './adapters/java.js';
   import './adapters/cpp.js';

   let initialized = false;

   async function ensureInit(): Promise<void> {
     if (initialized) return;
     await initParser();
     initialized = true;
   }

   async function getPreparedAdapter(languageId: LanguageId | null, relativePath: string): Promise<AstAdapter | null> {
     if (!languageId) return null;
     const adapter = getAdapter(languageId);
     if (!adapter) return null;
     // Adapters expose an optional prepare(). Cast via unknown-then-narrow.
     const maybePrep = (adapter as unknown as { prepare?: (rel?: string) => Promise<void> }).prepare;
     if (typeof maybePrep === 'function') {
       try { await maybePrep.call(adapter, relativePath); }
       catch { /* prepare failure — adapter unusable; return null so caller falls back */ return null; }
     }
     return adapter;
   }

   async function handle(payload: AnalyzePayload): Promise<AnalysisResponse> {
     try {
       await ensureInit();
       const changedSymbolsPerFile = new Map<string, { pre: SymbolIndex; post: SymbolIndex }>();
       const unsupported = new Set<string>();
       for (const file of payload.changedFiles) {
         // Defense-in-depth: parent path-validated already; worker re-checks size via shouldSkip.
         if (shouldSkip(file.relativePath)) continue;
         const lang = file.languageId ?? detectLanguageFromPath(file.relativePath);
         const adapter = await getPreparedAdapter(lang, file.relativePath);
         if (!adapter) {
           unsupported.add(lang ?? 'unknown');
           continue;
         }
         // Per-file try/catch — one bad file must not poison the pass.
         try {
           const pre = file.preContent !== null ? adapter.extractSymbols(file.preContent, file.relativePath)
             : { functions: [], classes: [], variables: [], imports: [], exports: [] };
           const post = file.postContent !== null ? adapter.extractSymbols(file.postContent, file.relativePath)
             : { functions: [], classes: [], variables: [], imports: [], exports: [] };
           changedSymbolsPerFile.set(file.relativePath, { pre, post });
         } catch (err) {
           console.error('[ast-worker] extractSymbols failed for', file.relativePath, err);
           unsupported.add(lang ?? 'unknown');
         }
       }

       const memberRefs = new Map<string, Map<string, ReferenceIndex>>();
       for (const [memberId, files] of Object.entries(payload.memberTrackedFiles)) {
         const fileMap = new Map<string, ReferenceIndex>();
         for (const f of files) {
           if (shouldSkip(f.relativePath)) continue;
           const lang = f.languageId ?? detectLanguageFromPath(f.relativePath);
           const adapter = await getPreparedAdapter(lang, f.relativePath);
           if (!adapter) continue;
           try {
             fileMap.set(f.relativePath, adapter.extractReferences(f.content, f.relativePath));
           } catch (err) {
             console.error('[ast-worker] extractReferences failed for', f.relativePath, err);
           }
         }
         if (fileMap.size > 0) memberRefs.set(memberId, fileMap);
       }

       const result = joinImpact(
         changedSymbolsPerFile,
         memberRefs,
         new Map(Object.entries(payload.memberDisplayNames)),
         [...unsupported].sort(),
       );
       return { requestId: payload.requestId, ok: true, result };
     } catch (err) {
       const message = err instanceof Error ? err.message : String(err);
       return { requestId: payload.requestId, ok: false, error: message };
     }
   }

   process.on('message', (msg: AnalyzePayload) => {
     // Run async, send response when ready.
     handle(msg).then(resp => {
       if (process.send) process.send(resp);
     }).catch(err => {
       if (process.send) process.send({ requestId: msg.requestId, ok: false, error: String(err) } satisfies AnalysisResponse);
     });
   });

   process.on('uncaughtException', (err) => {
     // Last-resort: notify parent (best-effort) then exit. Parent re-forks lazily.
     console.error('[ast-worker] uncaughtException', err);
     try { if (process.send) process.send({ requestId: '', ok: false, error: 'uncaughtException' } satisfies AnalysisResponse); } catch { /* ignore */ }
     process.exit(1);
   });

   process.on('disconnect', () => process.exit(0));
   ```

   import * as path from 'node:path';
   import * as crypto from 'node:crypto';
   import type {
     AnalyzePayload, AnalysisResponse, AnalysisResult,
     LanguageId,
   } from './types.js';
   import { detectLanguageFromPath } from './AstFactory.js';

   const DEFAULT_TIMEOUT_MS = 5000;          // T-05-02
   const CIRCUIT_BREAK_AFTER = 3;            // T-05-01 strikes
   const CIRCUIT_COOLDOWN_MS = 30_000;

   type PendingRequest = {
     resolve: (resp: AnalysisResponse) => void;
     timer: NodeJS.Timeout;
   };

   export interface AstAnalyzerOptions {
     timeoutMs?: number;
     workerScriptPath?: string;   // test override — point at dist/ast-worker.js by default
     now?: () => number;          // test clock override
   }

   export class AstAnalyzer {
     private worker: ChildProcess | null = null;
     private pending = new Map<string, PendingRequest>();
     private consecutiveFailures = 0;
     private circuitOpenUntil = 0;
     private readonly timeoutMs: number;
     private readonly workerScript: string;
     private readonly now: () => number;

     constructor(
       private readonly workspaceRoot: string,
       private readonly branchDir: string,
       opts: AstAnalyzerOptions = {},
     ) {
       this.timeoutMs = opts.timeoutMs ?? DEFAULT_TIMEOUT_MS;
       this.workerScript = opts.workerScriptPath ?? path.join(__dirname, '..', 'ast-worker.js');
       this.now = opts.now ?? Date.now;
     }

     async analyzeChange(args: {
       changedFiles: AnalyzePayload['changedFiles'];
       memberTrackedFiles: AnalyzePayload['memberTrackedFiles'];
       memberDisplayNames: AnalyzePayload['memberDisplayNames'];
     }): Promise<AnalysisResult> {
       const emptyResult: AnalysisResult = { affectedSymbols: [], perMember: {}, unsupportedLanguages: [] };
       if (this.circuitOpenUntil > this.now()) return emptyResult;

       const safe = this.validateAndFilter(args);
       if (safe.changedFiles.length === 0) return emptyResult;

       const worker = this.ensureWorker();
       if (!worker) return emptyResult;

       const requestId = crypto.randomUUID();
       const payload: AnalyzePayload = {
         requestId,
         changedFiles: safe.changedFiles,
         memberTrackedFiles: safe.memberTrackedFiles,
         memberDisplayNames: args.memberDisplayNames,
       };

       return new Promise<AnalysisResult>((resolve) => {
         const timer = setTimeout(() => {
           this.pending.delete(requestId);
           this.killAndMark();
           resolve(emptyResult);
         }, this.timeoutMs);

         this.pending.set(requestId, {
           timer,
           resolve: (resp) => {
             clearTimeout(timer);
             if (resp.ok) {
               this.consecutiveFailures = 0;
               resolve(resp.result);
             } else {
               this.markFailure();
               resolve(emptyResult);
             }
           },
         });
         worker.send(payload);
       });
     }

     dispose(): void {
       if (this.worker) {
         this.worker.kill('SIGTERM');
         this.worker = null;
       }
       for (const p of this.pending.values()) clearTimeout(p.timer);
       this.pending.clear();
     }

     private ensureWorker(): ChildProcess | null {
       if (this.worker) return this.worker;
       try {
         this.worker = fork(this.workerScript, [], { silent: false });
       } catch (err) {
         console.error('[AstAnalyzer] fork failed', err);
         this.markFailure();
         return null;
       }
       this.worker.on('message', (msg: AnalysisResponse) => {
         const p = this.pending.get(msg.requestId);
         if (p) {
           this.pending.delete(msg.requestId);
           p.resolve(msg);
         }
       });
       this.worker.on('exit', () => {
         // Settle all pending with empty / failure.
         for (const [id, p] of this.pending) {
           p.resolve({ requestId: id, ok: false, error: 'worker exited' });
         }
         this.pending.clear();
         this.worker = null;
         this.markFailure();
       });
       this.worker.on('error', (err) => {
         console.error('[AstAnalyzer] worker error', err);
       });
       return this.worker;
     }

     private killAndMark(): void {
       if (this.worker) {
         this.worker.kill('SIGTERM');
         this.worker = null;
       }
       this.markFailure();
     }

     private markFailure(): void {
       this.consecutiveFailures += 1;
       if (this.consecutiveFailures >= CIRCUIT_BREAK_AFTER) {
         this.circuitOpenUntil = this.now() + CIRCUIT_COOLDOWN_MS;
       }
     }

     // T-05-03 mitigation: drop unsafe paths.
     private validateAndFilter(args: {
       changedFiles: AnalyzePayload['changedFiles'];
       memberTrackedFiles: AnalyzePayload['memberTrackedFiles'];
       memberDisplayNames: AnalyzePayload['memberDisplayNames'];
     }): { changedFiles: AnalyzePayload['changedFiles']; memberTrackedFiles: AnalyzePayload['memberTrackedFiles'] } {
       const isSafe = (rel: string): boolean => {
         if (typeof rel !== 'string' || rel.length === 0 || rel.length > 1024) return false;
         if (rel.includes('\\')) return false;
         if (rel.startsWith('/')) return false;
         if (/^[A-Za-z]:[\\/]/.test(rel)) return false;
         const segments = rel.split('/');
         if (segments.includes('..')) return false;
         return true;
       };
       const cf = args.changedFiles.filter(f => isSafe(f.relativePath));
       const mtf: AnalyzePayload['memberTrackedFiles'] = {};
       for (const [memberId, files] of Object.entries(args.memberTrackedFiles)) {
         const filtered = files.filter(f => isSafe(f.relativePath));
         if (filtered.length > 0) mtf[memberId] = filtered;
       }
       return { changedFiles: cf, memberTrackedFiles: mtf };
     }
   }
   ```
