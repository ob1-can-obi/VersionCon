       private async runAstAnalysisAndAmend(
         recordId: string,
         pushRecord: PushRecord,
         prePostByFile: Map<string, { preContent: string | null; postContent: string | null }>,
       ): Promise<void> {
         try {
           if (!this.astAnalyzer) return;

           // Build the changedFiles payload from prePostByFile.
           const changedFiles = [...prePostByFile.entries()].map(([relativePath, { preContent, postContent }]) => ({
             relativePath,
             preContent,
             postContent,
             languageId: null,   // analyzer.analyzeChange's worker derives via detectLanguageFromPath
           }));

           // Build memberTrackedFiles by reading content from the branch dir.
           // Important: the host process has access to the entire branch tree
           // (it IS the branch source-of-truth), so we read tracked-paths
           // content from there rather than over the wire.
           const memberTrackedFiles: Record<string, Array<{ relativePath: string; content: string; languageId: null }>> = {};
           const memberDisplayNames: Record<string, string> = {};
           const names = this.getMemberNames();
           for (const [memberId, trackedPaths] of this.memberTracking) {
             // Exclude the pusher — they're the source of the change, not a caller.
             if (memberId === pushRecord.memberId) continue;
             const files: Array<{ relativePath: string; content: string; languageId: null }> = [];
             for (const rel of trackedPaths) {
               try {
                 // T-05-04 size cap is enforced by AstAnalyzer.validateAndFilter +
                 // skipPolicy in the worker; here we just stat + skip-if-large to
                 // avoid wasting IPC bandwidth on files the worker would skip anyway.
                 const stat = await fs.stat(path.join(this.branchDirGetter(), rel));
                 if (stat.size > 500_000) continue;
                 const content = await fs.readFile(path.join(this.branchDirGetter(), rel), 'utf-8');
                 files.push({ relativePath: rel, content, languageId: null });
               } catch { /* file missing in branch — skip */ }
             }
             if (files.length > 0) {
               memberTrackedFiles[memberId] = files;
               memberDisplayNames[memberId] = names.get(memberId) ?? 'Unknown';
             }
           }

           const result = await this.astAnalyzer.analyzeChange({
             changedFiles, memberTrackedFiles, memberDisplayNames,
           });

           if (result.affectedSymbols.length === 0 && result.unsupportedLanguages.length === 0) {
             // Nothing to amend — older message stands.
             return;
           }

           // Broadcast the amend. Host process echoes via ChatLog patch too so
           // chat-history replay carries the amended meta.
           const amendTs = createTimestamp();
           if (this.chatLog) {
             try {
               await this.chatLog.patchMeta(recordId, {
                 affectedSymbols: result.affectedSymbols,
                 unsupportedLanguages: result.unsupportedLanguages,
               });
             } catch (err) {
               console.error('[SessionHost] chat-log patchMeta failed', err);
             }
           }
           this.broadcast({
             type: 'chat-message-amend',
             timestamp: amendTs,
             recordId,
             affectedSymbols: result.affectedSymbols,
             unsupportedLanguages: result.unsupportedLanguages,
           });
           this.emit('chat-message-amend', {
             recordId,
             affectedSymbols: result.affectedSymbols,
             unsupportedLanguages: result.unsupportedLanguages,
           });
         } catch (err) {
           console.error('[SessionHost] runAstAnalysisAndAmend failed', err);
         }
       }

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | DoS via worker crash | SessionHost.runAstAnalysisAndAmend try/catch | mitigate | The fire-and-forget call is wrapped in try/catch at the top level; AstAnalyzer itself handles re-fork + 3-strike circuit (Plan 05-04 mitigation).
