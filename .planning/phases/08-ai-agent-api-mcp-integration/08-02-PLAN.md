```typescript
//   set[A-Z] | push[A-Z] | update[A-Z] | delete[A-Z] | commit[A-Z]

import type { BranchInfo } from '../types/branch.js';   // adapt path if BranchInfo lives elsewhere
import type { PushRecord } from '../types/push.js';
import type { ChatRecord } from '../types/chat.js';
import type { PresenceInfo } from '../types/session.js';

export interface BranchReader {
  getActiveBranch(): Promise<string>;
  listBranches(): readonly BranchInfo[];
}

export interface SyncReader {
  getOutOfSyncPaths(): readonly string[];
  getLatestPushId(): string | null;
}

export interface ActivityReader {
  getRecentPushes(limit: number): readonly PushRecord[];

export interface ChatReader {
  getRecent(limit: number): readonly ChatRecord[];

  forwardDeps(target: string, hops: 1 | 2): Promise<{ symbols: string[]; files: string[] }>;
  reverseDeps(target: string, hops: 1 | 2): Promise<{ symbols: string[]; files: string[] }>;
}

export interface PresenceReader {

}
```

```typescript
// and the six *ReaderImpl.ts adapters. Source-grep gates per CONTEXT

import * as assert from 'node:assert';
import * as fs from 'node:fs';
import * as fsAsync from 'node:fs/promises';
import * as path from 'node:path';

const REPO_ROOT = path.resolve(__dirname, '../../../..');
const MCP_DIR = path.join(REPO_ROOT, 'src', 'mcp');
const READERS_PATH = path.join(MCP_DIR, 'readers.ts');

async function readAllMcpFiles(): Promise<Array<{ path: string; text: string }>> {
  const out: Array<{ path: string; text: string }> = [];
  async function walk(dir: string): Promise<void> {
    let entries: fs.Dirent[];
    try {
      entries = await fsAsync.readdir(dir, { withFileTypes: true });
    } catch { return; }
    for (const e of entries) {
      const full = path.join(dir, e.name);
      if (e.isDirectory()) {
        await walk(full);
      } else if (e.isFile() && (e.name.endsWith('.ts') || e.name.endsWith('.js'))) {
        out.push({ path: full, text: await fsAsync.readFile(full, 'utf-8') });
      }
    }
  }
  await walk(MCP_DIR);
  return out;
}
```

```typescript
// src/mcp/adapters/DependencyReaderImpl.ts
// AD-HOC per-call dep query (RESEARCH §I.1 + Pitfall 7).
// v1 is single-file analysis only — NO workspace-wide walk, NO standing index.
// Defer the standing index to Phase 8.1 if per-call latency bites.
//
// forwardDeps(target):
//   - if target is a file path: call AstFactory.detectLanguageFromPath -> getAdapter
//     -> ast.<extract>(fileContents) -> return the referenced symbols + their files
//   - if target is a symbol name: scan workspace (capped) for the symbol's
//     definition file, then proceed as above. In v1 we cap this to a no-op
//     and return `{ symbols: [], files: [] }` when the target isn't a path
//     (symbols-as-entry-point is a 8.1 feature).
//
// reverseDeps(target):
//   - For a file: open the file, extract its EXPORTED symbols, then
//     return the set of files that import any of those symbols. v1 returns
//     `{ symbols: [], files: [] }` for symbol-entry-point inputs.
//
// All paths are workspace-relative. Defensive: missing file / unsupported
// language returns `{ symbols: [], files: [] }` (no throw).
import type { DependencyReader } from '../readers.js';
import * as fs from 'node:fs/promises';
import * as path from 'node:path';
import { detectLanguageFromPath, getAdapter } from '../../ast/AstFactory.js';
// `joinImpact` is the existing Phase 5 join helper — used here for the
// symbol→files inversion. If joinImpact requires a different input shape than
// we have here, the executor adapts the call site; the public DependencyReader
// contract stays the same.
// import { joinImpact } from '../../ast/joinImpact.js';   // import as needed

export interface DependencyReaderDeps {
  /** Absolute path to the workspace root. */
  workspaceRoot: string;
}

export class DependencyReaderImpl implements DependencyReader {
  constructor(private readonly deps: DependencyReaderDeps) {}

  async forwardDeps(
    target: string,
    _hops: 1 | 2,
  ): Promise<{ symbols: string[]; files: string[] }> {
    // v1: only file-path entry points are supported. Symbol-entry-point
    // resolution requires a standing index — deferred to 8.1.
    if (!looksLikeFilePath(target)) {
      return { symbols: [], files: [] };
    }
    const lang = detectLanguageFromPath(target);
    if (!lang) return { symbols: [], files: [] };
    const adapter = getAdapter(lang);
    if (!adapter) return { symbols: [], files: [] };
    const absPath = path.isAbsolute(target) ? target : path.join(this.deps.workspaceRoot, target);
    let source: string;
    try {
      source = await fs.readFile(absPath, 'utf-8');
    } catch {
      return { symbols: [], files: [] };
    }
    // Call the adapter's reference-extraction. The exact method name lives in
    // src/ast/types.ts — `extractReferences` is the assumed name; if the
    // actual method is `analyze` or `extract` adapt at the executor's read time.
    // Each adapter is expected to return an object with `references: string[]`
    // and optionally `referencedFiles: string[]` — see src/ast/types.ts AstAdapter.
    const result = await adapter.extractReferences(source, target);
    // v1 1-hop slice: take the direct references.
    return {
      symbols: [...(result?.references ?? [])].slice(0, 100),
      files:   [...(result?.referencedFiles ?? [])].slice(0, 100),
    };
  }

  async reverseDeps(
    target: string,
    _hops: 1 | 2,
  ): Promise<{ symbols: string[]; files: string[] }> {
    // v1: symbol-entry-point reverse lookup requires a standing index.
    // For a file-path entry point, returning the empty set is acceptable v1
    // behavior; 08-06/07 tools document this limitation in their tool
    // description ("Reverse deps in v1 are best-effort; defer to 8.1 standing
    // index for full coverage").
    if (!looksLikeFilePath(target)) {
      return { symbols: [], files: [] };
    }
    // 1-hop reverse for a file: in v1, return an empty set with no error.
    // The DependencyReader contract is satisfied (no throw, well-typed shape).
    return { symbols: [], files: [] };
  }
}

function looksLikeFilePath(s: string): boolean {
  return /[/\\.]/.test(s) && /\.(ts|tsx|js|jsx|py|java|cc?|cpp|hh?|hpp|mjs|cjs)$/i.test(s);
}
```
