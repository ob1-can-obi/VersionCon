// src/mcp/tools/adviseSync.ts
//
// Phase 8 — composite advise_sync tool (AI-04 / SC-4).
//
// Fans into 4 readers (Sync + Presence + Dependency + Activity) and fuses
// their signals into a calibrated confidence-scored prediction list per
// RESEARCH §I.2-I.3.
//
// Pure-fn design: handler reads via readers, then delegates the actual
// signal fusion to `fusePredictedConflicts(...)`. This keeps the heuristic
// unit-testable without spinning up an MCP server.
import * as z from 'zod/v4';   // adapt to 'zod' if needed
import type { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
import type { CallToolResult } from '@modelcontextprotocol/sdk/types.js';
import { registerReadOnlyTool } from '../registry.js';
import type {
  SyncReader, PresenceReader, DependencyReader, ActivityReader,
} from '../readers.js';

export type ConflictReason = 'ast-symbol-overlap' | 'file-edit-overlap' | 'lock-held-by-peer';

export interface PredictedConflict {
  file: string;
  reason: ConflictReason;
  confidence: number;
  detail: string;
  peer?: string;
}

export interface AdviseSyncState {
  behind: number;
  ahead: number;
  dirty: string[];
  last_sync_at: string | null;
}

export interface AdviseSyncPayload {
  state: AdviseSyncState;
  predicted_conflicts: PredictedConflict[];
}

/**
 * Pure function fusing the three signal sources into a deduped, confidence-
 * scored prediction list. RESEARCH §I.3 calibration:
 *   0.9 — peer has file open AND user editing same file (file-edit-overlap)
 *   0.7 — peer pushed symbol S, user references S (ast-symbol-overlap)
 *   0.6 — peer pushed file F, user has F open dirty (file-edit-overlap)
 *   0.5 — user edited symbol S, peer's open file imports S (ast-symbol-overlap)
 *   0.2 — behind ≥1 push with no other signal (generic out-of-sync)
 *
 * Inputs are deliberately narrow shapes (not full types) so this fn can be
 * unit-tested without spinning up the readers.
 */
export function fusePredictedConflicts(args: {
  dirtyFiles: readonly string[];
  behind: number;
  presenceByFile: ReadonlyMap<string, { memberId: string; displayName?: string }>;
  recentPushedFiles: readonly { file: string; actor: string }[];
  userReferences: ReadonlyMap<string, readonly { symbol: string; file: string }[]>;
  // userReferences: dirty file → array of symbols+files the user's code in that file uses
  peerPushedSymbols: readonly { symbol: string; actor: string }[];
}): PredictedConflict[] {
  const out: PredictedConflict[] = [];
  const seen = new Set<string>();   // dedup key: file|reason|peer

  function addUnique(p: PredictedConflict): void {
    const key = `${p.file}|${p.reason}|${p.peer ?? ''}`;
    if (seen.has(key)) return;
    seen.add(key);
    out.push(p);
  }

  // Tier 0.9: peer has file open AND user is editing same file
  for (const dirty of args.dirtyFiles) {
    const peer = args.presenceByFile.get(dirty);
    if (peer) {
      addUnique({
        file: dirty,
        reason: 'file-edit-overlap',
        confidence: 0.9,
        detail: `${peer.displayName ?? peer.memberId} has ${dirty} open while you are editing it`,
        peer: peer.displayName ?? peer.memberId,
      });
    }
  }

  // Tier 0.7: peer pushed a symbol that the user's dirty file references
  for (const dirty of args.dirtyFiles) {
    const refs = args.userReferences.get(dirty) ?? [];
    for (const ref of refs) {
      const peerPush = args.peerPushedSymbols.find(p => p.symbol === ref.symbol);
      if (peerPush) {
        addUnique({
          file: dirty,
          reason: 'ast-symbol-overlap',
          confidence: 0.7,
          detail: `${peerPush.actor} pushed a change to symbol ${ref.symbol}; you reference it in ${dirty}`,
          peer: peerPush.actor,
        });
      }
    }
  }

  // Tier 0.6: peer pushed a file that user has open dirty
  for (const dirty of args.dirtyFiles) {
    const peerFile = args.recentPushedFiles.find(p => p.file === dirty);
    if (peerFile) {
      addUnique({
        file: dirty,
        reason: 'file-edit-overlap',
        confidence: 0.6,
        detail: `${peerFile.actor} recently pushed ${dirty}; you have it open and dirty`,
        peer: peerFile.actor,
      });
    }
  }

  // Tier 0.5: user edited a symbol S in their dirty file; peer's open file
  // imports S. Heuristic: for each ref in userReferences for a dirty file
  // (i.e. user's outgoing refs), if presenceByFile mentions a peer file
  // that's the SAME ref.file, flag it.
  for (const dirty of args.dirtyFiles) {
    const refs = args.userReferences.get(dirty) ?? [];
    for (const ref of refs) {
      const peerOnRefFile = args.presenceByFile.get(ref.file);
      if (peerOnRefFile && !out.some(o => o.file === ref.file && o.peer === (peerOnRefFile.displayName ?? peerOnRefFile.memberId))) {
        addUnique({
          file: ref.file,
          reason: 'ast-symbol-overlap',
          confidence: 0.5,
          detail: `You edited symbol ${ref.symbol} in ${dirty}; ${peerOnRefFile.displayName ?? peerOnRefFile.memberId} has ${ref.file} (which imports ${ref.symbol}) open`,
          peer: peerOnRefFile.displayName ?? peerOnRefFile.memberId,
        });
      }
    }
  }

  // Tier 0.2: generic out-of-sync (behind > 0) with no other signals.
  // Only add if we have NO other predictions and behind > 0 (avoid noise).
  if (out.length === 0 && args.behind > 0) {
    out.push({
      file: '',
      reason: 'file-edit-overlap',
      confidence: 0.2,
      detail: `You are behind by ${args.behind} push${args.behind === 1 ? '' : 'es'}; consider syncing.`,
    });
  }

  return out;
}

export function registerAdviseSync(
  server: McpServer,
  deps: {
    syncReader: SyncReader;
    presenceReader: PresenceReader;
    depReader: DependencyReader;
    activityReader: ActivityReader;
  },
): void {
  registerReadOnlyTool(server, 'advise_sync', {
    title: 'Sync Advice',
    description:
      "Returns the user's current sync state (behind/ahead/dirty) plus a list of predicted " +
      "conflicts with confidence scores, sourced from the active VersionCon session's " +
      "presence tracking and AST dependency graph. Call this before suggesting a push, " +
      "before running code, or when asked 'am I in sync?'. Read-only — never blocks or " +
      "mutates state. Scoped to the local user's view.",
    inputSchema: {
      target_files: z.array(z.string()).optional()
        .describe('Optional workspace-relative file paths to scope the conflict prediction. ' +
          'When absent, scopes to all dirty + recently-edited files (typical case). ' +
          'Pass an empty array to get state-only with no predictions.'),
    },
  }, async ({ target_files }): Promise<CallToolResult> => {
    // Read all signals in parallel
    const dirty = [...deps.syncReader.getOutOfSyncPaths()];
    const latestPushes = deps.activityReader.getRecentPushes(20);
    const presence = deps.presenceReader.getPresenceSnapshot();

    // Build per-file presence map
    const presenceByFile = new Map<string, { memberId: string; displayName?: string }>();
    for (const p of presence) {
      // PresenceInfo shape: { memberId, displayName, activeFilePath, ... }
      if ((p as any).activeFilePath) {
        presenceByFile.set((p as any).activeFilePath, {
          memberId: (p as any).memberId,
          displayName: (p as any).displayName,
        });
      }
    }

    // Recent pushed files (flattened from pushes)
    const recentPushedFiles: { file: string; actor: string }[] = [];
    const peerPushedSymbols: { symbol: string; actor: string }[] = [];
    for (const push of latestPushes) {
      for (const f of (push as any).files ?? []) {
        recentPushedFiles.push({ file: f, actor: (push as any).memberDisplayName });
      }
      // If PushRecord has meta.affectedSymbols (Phase 5 stamp), collect them.
      // FakeReaders' PushRecord may or may not have this; defensive read.
      const affected = ((push as any).meta?.affectedSymbols ?? []) as string[];
      for (const sym of affected) {
        peerPushedSymbols.push({ symbol: sym, actor: (push as any).memberDisplayName });
      }
    }

    // userReferences: for each dirty file, look up its forward deps
    // (single-file ad-hoc per Pitfall 7 — bounded by dirty.length which is
    // typically small)
    const userReferences = new Map<string, { symbol: string; file: string }[]>();
    for (const f of dirty) {
      const fwd = await deps.depReader.forwardDeps(f, 1);
      const refs: { symbol: string; file: string }[] = [];
      for (const s of fwd.symbols) refs.push({ symbol: s, file: '' });
      for (const fileRef of fwd.files) refs.push({ symbol: '', file: fileRef });
      userReferences.set(f, refs);
    }

    // Scope to target_files if provided
    let scopedDirty = dirty;
    if (target_files !== undefined) {
      if (target_files.length === 0) {
        // Empty array → state-only, skip predictions
        scopedDirty = [];
      } else {
        scopedDirty = dirty.filter(f => target_files.includes(f));
      }
    }

    const behind = (deps.syncReader.getLatestPushId() && latestPushes.length > 0
      && latestPushes[0]?.id !== deps.syncReader.getLatestPushId()) ? 1 : 0;
    const lastPush = latestPushes[0];
    const state: AdviseSyncState = {
      behind: scopedDirty.length === 0 && target_files?.length === 0 ? 0 : Math.max(scopedDirty.length, behind),
      ahead: 0,    // v1 placeholder per CONTEXT D-2 / RESEARCH note
      dirty: scopedDirty,
      last_sync_at: lastPush ? (lastPush as any).timestamp : null,
    };

    const predicted_conflicts = scopedDirty.length === 0 && target_files?.length === 0
      ? []
      : fusePredictedConflicts({
          dirtyFiles: scopedDirty,
          behind: state.behind,
          presenceByFile,
          recentPushedFiles,
          userReferences,
          peerPushedSymbols,
        });

    const payload: AdviseSyncPayload = { state, predicted_conflicts };
    return { content: [{ type: 'text', text: JSON.stringify(payload) }] };
  });
}
