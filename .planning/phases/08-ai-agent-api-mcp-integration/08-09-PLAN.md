// Add lazy MCP OutputChannel factory near getGitBridgeOutputChannel (line 189):
let mcpOutputChannel: vscode.OutputChannel | null = null;
let mcpChannelPushedToSubs = false;
function getMcpOutputChannel(context: vscode.ExtensionContext): vscode.OutputChannel {
  if (!mcpOutputChannel) mcpOutputChannel = vscode.window.createOutputChannel('VersionCon: MCP');
  if (!mcpChannelPushedToSubs) {
    context.subscriptions.push(mcpOutputChannel);
    mcpChannelPushedToSubs = true;
  }
  return mcpOutputChannel;
}

// Track the running MCP handle so deactivate() can close it
let runningMcpHandle: McpServerHandle | null = null;

// Inside activate(), AFTER session start wiring, OR in the
// onSessionStarted callback if MCP should follow session lifecycle:
//
// Variant A — start MCP on session start (recommended; needs session refs):
function startMcpForSession(ctx: vscode.ExtensionContext, sessionHost: SessionHost, branchManager: BranchManager, syncTracker: SyncTracker, pushHistory: PushHistory, chatLog: ChatLog): void {
  const folder = vscode.workspace.workspaceFolders?.[0]?.uri.fsPath;
  if (!folder) return;
  const log = (line: string) => getMcpOutputChannel(ctx).appendLine(line);
  void startMcpLifecycle({
    context: ctx,
    log,
    deps: {
      branchReader:   new BranchReaderImpl(branchManager),
      syncReader:     new SyncReaderImpl(syncTracker),
      activityReader: new ActivityReaderImpl(pushHistory),
      chatReader:     new ChatReaderImpl(chatLog),
      depReader:      new DependencyReaderImpl({ workspaceRoot: folder }),
      presenceReader: new PresenceReaderImpl(sessionHost),
    },
    ensureConsent,
    upsertMcpConfig: async (port: number) => {
      const url = `http://127.0.0.1:${port}/mcp`;
      await upsertMcpConfig(folder, '.vscode/mcp.json', 'versioncon', url);
      await upsertMcpConfig(folder, '.mcp.json', 'versioncon', url);
    },
    removeMcpConfig: async () => {
      await removeMcpConfig(folder, '.vscode/mcp.json', 'versioncon');
      await removeMcpConfig(folder, '.mcp.json', 'versioncon');
    },
  }).then((handle) => { runningMcpHandle = handle; })
    .catch((err) => log(`[mcp] startup failed: ${String((err as Error)?.message ?? err)}`));
}

// In deactivate():
export async function deactivate(): Promise<void> {
  if (runningMcpHandle) {
    try {
      await stopMcpLifecycle(runningMcpHandle, {
        removeMcpConfig: async () => {
          const folder = vscode.workspace.workspaceFolders?.[0]?.uri.fsPath;
          if (!folder) return;
          await removeMcpConfig(folder, '.vscode/mcp.json', 'versioncon');
          await removeMcpConfig(folder, '.mcp.json', 'versioncon');
        },
        log: (line) => { /* no OutputChannel after deactivate — write to nothing */ },
      });
    } catch { /* swallow — deactivate must not throw */ }
    runningMcpHandle = null;
  }
}

## AI Agents (MCP integration)

VersionCon ships a local Model Context Protocol (MCP) server that exposes the active session's collab state — branch, sync, recent activity, chat log, and the dependency graph — as **read-only** tools to AI coding agents.

**Supported clients out-of-the-box:**
- **VS Code Copilot** (1.95+) reads `.vscode/mcp.json` automatically.
- **Claude Code** reads `.mcp.json` at the workspace root.
